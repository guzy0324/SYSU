linear = set()
for feature in (4, 5, 6, 16, 19, 20, 22):
    linear.add(feature - 1)

miss = dict()
mode = dict()
domain = dict()
training_inputs = list()
training_outputs = list()
with open('horse-colic.data') as f:
    while True:
        line = f.readline()
        if not line:
            break
        e = line.split()[:23]
        last = e.pop()
        if last == '?':
            continue
        cl = int(last) >= 2
        training_outputs.append([float(i == cl) for i in range(2)])
        for k, v in enumerate(e):
            if v == '?':
                if len(training_inputs) not in miss:
                    miss[len(training_inputs)] = list()
                miss[len(training_inputs)].append(k)
                mode[k] = None
            else:
                e[k] = float(v)
                if k not in domain:
                    domain[k] = [float('inf'), -float('inf')]
                domain[k][0] = min(e[k], domain[k][0])
                domain[k][1] = max(e[k], domain[k][1])
        training_inputs.append(e)

for example in training_inputs:
    for k in mode:
        v = example[k]
        if v != '?':
            if k in linear:
                if mode[k] is None:
                    mode[k] = [0, 0]
                mode[k][0] += v
                mode[k][1] += 1
            else:
                if mode[k] is None:
                    mode[k] = dict()
                if v not in mode[k]:
                    mode[k][v] = 0
                mode[k][v] += 1

for k, v in mode.items():
    if k in linear:
        mode[k] = v[0] / v[1];
    else:
        mode[k] = max(v, key=lambda x: v[x])

for k, v in miss.items():
    for i in v:
        training_inputs[k][i] = mode[i]

for example in training_inputs:
    for k, v in enumerate(example):
        example[k] = (v - domain[k][0]) / (domain[k][1] - domain[k][0])

def cal_acc_f1(nn, testing_sets):
    count = [[0, 0, 0, 0], [0, 0, 0, 0]]
    #TP, FP, TN, FN
    for inputs, outputs in testing_sets:
        predicts = nn.feed_forward(inputs)
        predict = max(range(len(predicts)), key=lambda x: predicts[x])
        output = max(range(len(outputs)), key=lambda x: outputs[x])
        for i in range(2):
            if i == output:
                if predict == output:
                    count[i][0] += 1
                else:
                    count[i][1] += 1
            else:
                if predict == output:
                    count[i][2] += 1
                else:
                    count[i][3] += 1
    accuracy = sum(count[0][::2]) / sum(count[0])

    f1 = 0
    for i in range(2):
        f1 += 2 * count[i][0] / (2 * count[i][0]
                                 + count[i][1] + count[i][3])
    f1 /= 2
    return accuracy, f1


def test(nn, training_sets, training_error, testing_error, training_accuracy, testing_accuracy, training_f1, testing_f1):
    training_error.append(nn.calculate_total_error(training_sets))
    testing_error.append(nn.calculate_total_error(testing_sets))
    acc, f1 = cal_acc_f1(nn, training_sets)
    training_accuracy.append(acc)
    training_f1.append(f1)
    acc, f1 = cal_acc_f1(nn, testing_sets)
    testing_accuracy.append(acc)
    testing_f1.append(f1)


def train_and_test():
    global SEED, LR_FUNC, EPOCH, HIDDEN, training_inputs, training_outputs, testing_sets
    random.seed(SEED)
    nn = NeuralNetwork(22, HIDDEN, 2)

    training_sets = list(zip(training_inputs, training_outputs))
    training_error = list()
    testing_error = list()
    training_accuracy = list()
    testing_accuracy = list()
    training_f1 = list()
    testing_f1 = list()

    test(nn, training_sets, training_error, testing_error, training_accuracy, testing_accuracy, training_f1, testing_f1)
    for epoch in tqdm(range(1, EPOCH + 1)):
        NeuralNetwork.LEARNING_RATE = LR_FUNC(epoch)
        for i in range(len(training_inputs)):
            nn.train(training_inputs[i], training_outputs[i])
        test(nn, training_sets, training_error, testing_error, training_accuracy, testing_accuracy, training_f1, testing_f1)

    plt.subplots_adjust(right=2, top=2)
    plt.subplot(2, 1, 1)
    plt.xlabel('epoch')
    plt.ylabel('error')
    plt.plot(range(EPOCH + 1), training_error)
    plt.plot(range(EPOCH + 1), testing_error)
    plt.legend(['train', 'test'])
    plt.subplot(2, 2, 3)
    plt.xlabel('epoch')
    plt.ylabel('accuracy')
    plt.plot(range(EPOCH + 1), training_accuracy)
    plt.plot(range(EPOCH + 1), testing_accuracy)
    plt.legend(['train', 'test'])
    plt.subplot(2, 2, 4)
    plt.xlabel('epoch')
    plt.ylabel('f1')
    plt.plot(range(EPOCH + 1), training_f1)
    plt.plot(range(EPOCH + 1), testing_f1)
    plt.legend(['train', 'test'])
